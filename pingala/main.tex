\section{JEE 2019}
\noindent Let $\alpha$ and $\beta$ ($\alpha > \beta$) be the roots of the equation $z^2-z-1=0$. Define,

\begin{align}
	a_n & = \frac{\alpha^n-\beta^n}{\alpha-\beta}, \quad n \ge 1\\
	b_n & = a_{n-1} - a_{n+1}, \quad n \ge 2, \quad b_1 = 1
\end{align}
